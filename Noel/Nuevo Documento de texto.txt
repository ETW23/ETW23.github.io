
    <style>
      .progress-bar {
        width: 500px;
        height: 20px;
        position: relative;
        margin: 50px auto;
        background: linear-gradient(
          to right,
          #00cc00 0%,
          /* Green up to 10% */ #ffff00 15%,

          /* Blend of yellow and orange */ #ff6600 45%,
          /* Orange from 40% to 50% */ #ff0000 55%,
          /* Blend of orange and red */ #800080 55%,
          /* Blend of red and purple */ #800080 60%,
          /* Purple from 60% */ #ff00ff 100% /* Vibrant pink at 100% */
        );
      }

      .progress {
        width: 100%;
        height: 100%;
        appearance: none; /* Remove default appearance */
      }

      .progress::-webkit-progress-bar {
        display: none; /* Remove default styling of the inner progress bar */
      }

      .circle {
        width: 20px;
        height: 20px;
        border-radius: 50%;
        background-color: #00cc00;
        position: absolute;
        border: 0.125rem solid black;
        top: -10px;
        cursor: pointer;
      }

      .number {
        position: absolute;
        top: -30px;
        left: 50%;
        transform: translateX(-50%);
        font-size: 14px;
      }





    <div class="progress-bar">
      <progress class="progress" value="0" max="20"></progress>
      <div class="circle"></div>
      <div class="number">0</div>
    </div>
    <script>
      const progress = document.querySelector(".progress");
      const circle = document.querySelector(".circle");
      const number = document.querySelector(".number");

      circle.addEventListener("mousedown", function (event) {
        event.preventDefault();
        let isDragging = true;

        document.addEventListener("mousemove", dragCircle);
        document.addEventListener("mouseup", function () {
          isDragging = false;
        });

        function dragCircle(event) {
          if (!isDragging) return;

          const progressRect = progress.getBoundingClientRect();
          const progressWidth = progressRect.width;
          let posX = event.clientX - progressRect.left;

          if (posX < 0) {
            posX = 0;
          } else if (posX > progressWidth) {
            posX = progressWidth;
          }

          const newValue = Math.round((posX / progressWidth) * 20);
          const newPosition = (newValue / 20) * 100;

          circle.style.left = `calc(${newPosition}% - 10px)`;

          progress.value = newValue;
          number.innerText = newValue;

          let color;
          if (newValue <= 2) {
            color = "#00cc00"; // Green
          } else if (newValue <= 5) {
            color = "#ffff00"; // Yellow
          } else if (newValue <= 7) {
            color = "#ff6600"; // Orange
          } else if (newValue <= 10) {
            color = "#ff0000"; // Red
          } else {
            color = "#800080"; // Purple for values above 11
          }

          circle.style.backgroundColor = color;
        }
      });
    </script>
